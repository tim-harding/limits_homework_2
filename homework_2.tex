\section*{2.1.10.a}

\subsection*{Problem}
Given $\epsilon = 0.314$, find $\delta$ such that $\forall x : \abs{x} < \delta \implies \abs{(4x + 7) - 7} < \epsilon$.

\subsection*{Solution}
\begin{align*}
    \abs{4x} &< \epsilon \\
    \abs{x} &< \frac{\epsilon}{4}
\end{align*}
We require $\delta = \frac{0.314}{4} = \boxed{0.0785}$.
